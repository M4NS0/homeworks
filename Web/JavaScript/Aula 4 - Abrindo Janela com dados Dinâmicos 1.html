Abrindo Janela com Dados Dinâmicos

<form name="form">
 <input type="text" name="nome">
 <button onclick="abrir()"> Abrir </button>
</form>

<script>
 function abrir() {
  janela = window.open("", "Dados", "height=700, width=400, scrollbar=yes");
  janela.document.write("<center> Presado Associado <b>" + document.form.nome.value + "</b></center>");
  janela.document.write("<center> Seja bem vindo(a)! </center>")
  janela.document.close();
 }
</script>
